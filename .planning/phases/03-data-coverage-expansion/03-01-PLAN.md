    ```python
    import logging
    import os

    import httpx

    from data_providers.rate_limiter import AsyncRateLimiter

    FINNHUB_BASE_URL = "https://finnhub.io/api/v1"
    logger = logging.getLogger(__name__)

    # Peer-basket proxy tickers for sector P/E derivation. Finnhub's free tier does
    # not expose sector aggregates directly, so we compute the median P/E of 3-5
    # well-known constituents per sector. Matches the sector keys in
    # agents/fundamental.py::SECTOR_PE_MEDIANS.
    _SECTOR_PROXY_TICKERS: dict[str, list[str]] = {
        "technology": ["AAPL", "MSFT", "GOOGL", "NVDA", "META"],
        "healthcare": ["JNJ", "UNH", "LLY", "MRK", "PFE"],
        "financial services": ["JPM", "BAC", "WFC", "GS", "MS"],
        "financials": ["JPM", "BAC", "WFC", "GS", "MS"],
        "consumer cyclical": ["AMZN", "TSLA", "HD", "NKE", "MCD"],
        "consumer defensive": ["WMT", "PG", "KO", "PEP", "COST"],
        "industrials": ["HON", "UNP", "CAT", "BA", "GE"],
        "energy": ["XOM", "CVX", "COP", "SLB", "EOG"],
        "utilities": ["NEE", "DUK", "SO", "AEP", "D"],
        "real estate": ["PLD", "AMT", "CCI", "EQIX", "SPG"],
        "basic materials": ["LIN", "SHW", "APD", "ECL", "FCX"],
        "communication services": ["GOOGL", "META", "NFLX", "DIS", "VZ"],
    }

    class FinnhubProvider(DataProvider):
        """Finnhub data provider for sector P/E, company metrics, insider trades, ESG.

        Free tier: 60 req/min, no daily cap. Non-commercial use only (solo-operator OK).
        """
        _limiter = AsyncRateLimiter(
            max_calls=int(os.getenv("FINNHUB_RATE_LIMIT", "60")),
            period_seconds=60.0,
        )

        def __init__(self, api_key: str | None = None, timeout: float = 10.0) -> None:
            resolved_key = api_key or os.getenv("FINNHUB_API_KEY")
            self._api_key = resolved_key
            self._timeout = timeout
            if not resolved_key:
                warnings.warn(
                    "FINNHUB_API_KEY not set. FinnhubProvider methods will raise RuntimeError.",
                    RuntimeWarning,
                )
                self._client: httpx.AsyncClient | None = None
            else:
                self._client = httpx.AsyncClient(
                    base_url=FINNHUB_BASE_URL,
                    params={"token": resolved_key},
                    timeout=timeout,
                )

        async def _rate_limited_get(self, path: str, params: dict[str, Any]) -> dict:
            if self._client is None:
                raise RuntimeError("Finnhub API key missing")
            async with self._limiter:
                try:
                    resp = await self._client.get(path, params=params)
                    resp.raise_for_status()
                    return resp.json()
                except httpx.HTTPStatusError as exc:
                    if exc.response.status_code == 429:
                        logger.warning("Finnhub rate limit hit for %s: %s", path, exc)
                        return {}
                    raise

        async def get_company_pe(self, ticker: str) -> float | None:
            """Return TTM trailing P/E for a single ticker, or None if unavailable."""
            data = await self._rate_limited_get(
                "/stock/metric", {"symbol": ticker, "metric": "all"}
            )
            metric = data.get("metric", {}) or {}
            pe = metric.get("peBasicExclExtraTTM") or metric.get("peNormalizedAnnual")
            if pe is None:
                return None
            try:
                pe_f = float(pe)
                if pe_f <= 0 or pe_f > 1000:  # sanity filter (negative earnings / extreme outliers)
                    return None
                return pe_f
            except (TypeError, ValueError):
                return None

        async def get_sector_pe(self, sector: str | None) -> float | None:
            """Compute median P/E of a hardcoded peer basket for the sector.

            Returns None if fewer than 2 peers return valid P/E (falls back to static
            SECTOR_PE_MEDIANS at the call site in agents/fundamental.py).
            """
            if not sector:
                return None
            key = sector.lower().strip()
            peers = _SECTOR_PROXY_TICKERS.get(key)
            if not peers:
                return None
            pes: list[float] = []
            for peer in peers:
                pe = await self.get_company_pe(peer)
                if pe is not None:
                    pes.append(pe)
            if len(pes) < 2:
                return None
            return float(statistics.median(pes))

        async def get_price_history(self, ticker, period="1y", interval="1d"):
            # Finnhub has /stock/candle but it's premium-tier only for historical
            # data. yfinance remains the OHLCV source; FinnhubProvider is metrics-only.
            raise NotImplementedError("FinnhubProvider does not provide OHLCV; use YFinanceProvider")

        async def get_current_price(self, ticker) -> float:
            data = await self._rate_limited_get("/quote", {"symbol": ticker})
            price = data.get("c")
            if price is None:
                raise ValueError(f"No Finnhub price for {ticker}")
            return float(price)

        def is_point_in_time(self) -> bool:
            return False

        def supported_asset_types(self) -> list[str]:
            return ["stock"]

        async def aclose(self) -> None:
            if self._client is not None:
                await self._client.aclose()
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Finnhub API <-> FinnhubProvider | Third-party HTTPS service; untrusted response body must be parsed defensively |
| Environment variable FINNHUB_API_KEY <-> Logs | Leakage risk: API key in stack traces, URL logging, or httpx debug output |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01-01 | Information Disclosure | FinnhubProvider | mitigate | `token` is passed as query-string param (httpx default). Add logging filter: never log `client.params` or full URLs at INFO+. Use `logger.warning("Finnhub rate limit hit for %s", path)` (path only, not params). |
| T-03-01-02 | Tampering | get_sector_pe response parsing | mitigate | Defensive parsing — coerce to float with try/except, drop values `<=0` or `>1000` as sanity filter, median(N>=2) requires consensus from multiple peers to reject a single poisoned value. |
