<objective>
Add Brier score and rolling Information Coefficient (IC/IC-IR) per agent, surface them via a new `GET /api/v1/analytics/calibration` endpoint, and feed the IC signal back into `engine/weight_adapter.py` so negative-IC agents lose weight on the next adaptation cycle. Reads from `backtest_signal_history` (populated by Plan 02-02) — not the 10-row live `signal_history`.

Purpose: Agent confidence today is self-reported; agent weights today come from raw hit rate. Without Brier we can't detect miscalibration (agent says 80% confidence but is right only 60% of the time). Without IC we can't detect an agent that is right often but with tiny magnitude (low-value) vs. one that's right less often but with high magnitude (high-value). This plan delivers the calibration data that Phase 4's WeightsPage will visualize.

Output:
- `tracking/store.py::get_backtest_signals_by_agent` — read backtest_signal_history filtered by agent
- `tracking/tracker.py::compute_brier_score` — one-vs-rest binary Brier on BUY/SELL only
- `tracking/tracker.py::compute_rolling_ic` — time-series Pearson IC with 60-obs rolling window
- `tracking/tracker.py::compute_icir` — mean(IC)/std(IC) over rolling window
- `engine/weight_adapter.py::compute_ic_weights` — IC-IR-based scaling factor; None → fallback to EWMA
- `api/routes/calibration.py` — new route; response includes per-agent metrics + corpus metadata
- Route registered under `/analytics` prefix in `api/app.py`
- 3 new test modules
</objective>

      ```python
      async def compute_brier_score(
          self,
          agent_name: str,
          horizon: str = "5d",
          min_samples: int = 20,
      ) -> float | None:
          """One-vs-rest binary Brier score for directional signals only.

          HOLD signals are excluded (AP-05).
          Returns None when N < min_samples (AP-03).
          Lower is better: 0.0 = perfect, 0.25 = random, 1.0 = perfectly wrong.
          """
          rows = await self._store.get_backtest_signals_by_agent(agent_name, horizon)
          directional = [
              r for r in rows
              if r["signal"] in ("BUY", "SELL") and r["forward_return"] is not None
          ]
          if len(directional) < min_samples:
              return None
          squared_errors = []
          for r in directional:
              prob = float(r["confidence"]) / 100.0  # normalize 0-100 → 0-1
              if r["signal"] == "BUY":
                  outcome = 1.0 if r["forward_return"] > 0 else 0.0
              else:  # SELL
                  outcome = 1.0 if r["forward_return"] < 0 else 0.0
              squared_errors.append((prob - outcome) ** 2)
          return round(sum(squared_errors) / len(squared_errors), 4)


      async def compute_rolling_ic(
          self,
          agent_name: str,
          horizon: str = "5d",
          window: int = 60,
          min_samples: int = 30,
      ) -> tuple[float | None, list[float | None]]:
          """Time-series Pearson IC + rolling IC series (SIG-03).

          Returns (overall_ic, rolling_ics). Both are None when N < min_samples.
          rolling_ics has one entry per position in the series; early positions
          with insufficient lookback are None (AP-03).
          """
          rows = await self._store.get_backtest_signals_by_agent(agent_name, horizon)
          # Use raw_score (continuous), NOT signal enum string (AP-02)
          scored = [
              (r["raw_score"], r["forward_return"])
              for r in rows
              if r["raw_score"] is not None and r["forward_return"] is not None
          ]
          if len(scored) < min_samples:
              return None, []

          from scipy.stats import pearsonr
          scores_all = [s for s, _ in scored]
          returns_all = [r for _, r in scored]

          # Overall IC: single Pearson across full series
          overall_ic, _ = pearsonr(scores_all, returns_all)
          overall_ic = float(overall_ic) if not (overall_ic != overall_ic) else None  # NaN guard

          # Rolling IC
          rolling: list[float | None] = []
          n = len(scored)
          for i in range(n):
              if i < window - 1:
                  rolling.append(None)
                  continue
              s_win = scores_all[i - window + 1 : i + 1]
              r_win = returns_all[i - window + 1 : i + 1]
              if len(s_win) < min_samples:
                  rolling.append(None)
                  continue
              try:
                  ic_val, _ = pearsonr(s_win, r_win)
                  rolling.append(float(ic_val) if not (ic_val != ic_val) else None)
              except Exception:
                  rolling.append(None)

          return (round(overall_ic, 4) if overall_ic is not None else None), rolling


      @staticmethod
      def compute_icir(rolling_ics: list[float | None]) -> float | None:
          """IC-IR = mean(IC) / std(IC) over rolling window.

          Returns None when < 5 valid IC values or std is zero (AP-03).
          """
          valid = [ic for ic in rolling_ics if ic is not None]
          if len(valid) < 5:
              return None
          import statistics
          mean_ic = statistics.mean(valid)
          std_ic = statistics.stdev(valid)
          if std_ic == 0:
              return None
          return round(mean_ic / std_ic, 4)
      ```

      ```python
          horizon: str = "5d",
          window: int = 60,
          scale_divisor: float = 2.0,
      ) -> AdaptiveWeights | None:
          """Compute per-agent weights from IC-IR (SIG-03).

          Scaling rule: `new_weight = base_weight * max(0, ic_ir / scale_divisor)`.
          Negative IC-IR → factor 0 → agent removed from aggregation (renormalized).
          Returns None if NO agents have sufficient data → caller falls back to EWMA.

          scale_divisor=2.0 rescales typical IC-IR values (O(0.3-1.0)) into O(0.15-0.5)
          weight-multiplier range. Configurable for future calibration.
          """
          if asset_types is None:
              asset_types = ["stock", "crypto"]
          if agents is None:
              agents = [
                  "TechnicalAgent", "FundamentalAgent", "MacroAgent",
                  "SentimentAgent", "CryptoAgent",
              ]

          weights: dict[str, dict[str, float]] = {at: {} for at in asset_types}
          any_valid = False
          total_sample_size = 0

          for agent in agents:
              overall_ic, rolling = await tracker.compute_rolling_ic(
                  agent, horizon=horizon, window=window,
              )
              icir = tracker.compute_icir(rolling) if rolling else None
              factor = max(0.0, icir / scale_divisor) if icir is not None else None
              sample_size = len([ic for ic in rolling if ic is not None]) if rolling else 0
              total_sample_size += sample_size

              if factor is None:
                  # Insufficient data — leave agent's weight at 0 (will be renormalized)
                  for at in asset_types:
                      weights[at][agent] = 0.0
                  continue
              any_valid = True
              for at in asset_types:
                  weights[at][agent] = factor

          if not any_valid:
              return None  # caller falls back to EWMA

          # Renormalize each asset_type's weights to sum to 1.0 (handle all-zero case)
          for at in asset_types:
              total = sum(weights[at].values())
              if total <= 0:
                  # All agents zero/negative — equal-weight fallback
                  equal = 1.0 / max(1, len(weights[at]))
                  weights[at] = {k: equal for k in weights[at]}
              else:
                  weights[at] = {k: v / total for k, v in weights[at].items()}

          from datetime import datetime, timezone
          return AdaptiveWeights(
              weights=weights,
              source="ic_ir",
              computed_at=datetime.now(timezone.utc).isoformat(),
              sample_size=total_sample_size,
              staleness_days=0,
          )
      ```
