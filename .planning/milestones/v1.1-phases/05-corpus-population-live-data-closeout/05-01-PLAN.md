<objective>
Ship the LIVE-01 backend surface: `POST /api/v1/calibration/rebuild-corpus` triggers
per-ticker signal-corpus rebuild for the user's portfolio (or explicit ticker list)
as a background task, and `GET /api/v1/calibration/rebuild-corpus/{job_id}` exposes
real-time progress. This unblocks Phase 6 CalibrationPage — until corpus rows exist
for the user's tickers, every calibration metric cell displays "insufficient data".

Purpose: Bridge the gap between v1.0's rebuild_signal_corpus job (proven per-ticker
FOUND-07 atomicity with DELETE rollback) and the HTTP surface Phase 6 will consume.
The endpoint is the thin orchestration layer: enumerate tickers, spawn background
task, persist progress, delegate per-ticker work to the already-correct daemon job.

Output:
- `POST /api/v1/calibration/rebuild-corpus` endpoint
- `GET /api/v1/calibration/rebuild-corpus/{job_id}` progress endpoint
- `corpus_rebuild_jobs` SQLite table with per-ticker progress JSON
- Pydantic request/response models in `api/models.py`
- Test suite asserting: background isolation, per-ticker failure isolation, progress polling, FOUND-07 contract preservation
</objective>

    ```python
    import asyncio
    import json
    import logging
    import uuid
    from datetime import datetime, timezone

    import aiosqlite
    from fastapi import BackgroundTasks, HTTPException

    from api.models import (
        RebuildCorpusRequest,
        RebuildCorpusResponse,
        RebuildCorpusProgressResponse,
    )

    _route_logger = logging.getLogger(__name__)


    @router.post("/calibration/rebuild-corpus", response_model=RebuildCorpusResponse)
    async def rebuild_corpus_endpoint(
        body: RebuildCorpusRequest,
        background_tasks: BackgroundTasks,
        db_path: str = Depends(get_db_path),
    ) -> RebuildCorpusResponse:
        """LIVE-01: Trigger backtest signal corpus rebuild.

        Body shape:
          {"tickers": ["AAPL","NVDA"] | null, "asset_types": {"AAPL":"stock"} | null}

        Returns 200 immediately with {job_id, status:"started", ticker_count}.
        Per-ticker work runs as an asyncio background task via FastAPI BackgroundTasks.
        Poll GET /calibration/rebuild-corpus/{job_id} for progress.

        Per-ticker isolation: each ticker is a separate rebuild_signal_corpus call,
        so one ticker failing does NOT abort the batch. Failures are captured in
        ticker_progress_json. The overall job status is:
          - "success"  if all tickers succeeded
          - "partial"  if some succeeded and some failed
          - "error"    if the batch itself errored before per-ticker work (rare)
        """
        from portfolio.manager import PortfolioManager

        # Resolve ticker list
        if body.tickers is None:
            pm = PortfolioManager(db_path)
            positions = await pm.get_all_positions()
            tickers_with_type: list[tuple[str, str]] = [
                (p.ticker, p.asset_type) for p in positions
            ]
        else:
            at_map = body.asset_types or {}
            tickers_with_type = [
                (t, at_map.get(t, "stock")) for t in body.tickers
            ]

        if not tickers_with_type:
            raise HTTPException(
                status_code=400,
                detail="No tickers to rebuild: portfolio empty and no explicit tickers supplied",
            )

        job_id = uuid.uuid4().hex
        started_at = datetime.now(timezone.utc).isoformat()

        # Persist initial job row
        async with aiosqlite.connect(db_path) as conn:
            await conn.execute(
                """
                INSERT INTO corpus_rebuild_jobs
                    (job_id, status, tickers_total, tickers_completed,
                     ticker_progress_json, started_at)
                VALUES (?, 'running', ?, 0, ?, ?)
                """,
                (
                    job_id,
                    len(tickers_with_type),
                    json.dumps({t: {"status": "pending"} for t, _ in tickers_with_type}),
                    started_at,
                ),
            )
            await conn.commit()

        # Schedule background task — FastAPI's BackgroundTasks runs AFTER response is sent.
        # For large batches we want isolation from the response lifecycle, so use
        # asyncio.create_task directly for robustness (BackgroundTasks is fine but
        # ties the task to the request scope, which matters less here).
        background_tasks.add_task(
            _run_batch_rebuild,
            db_path=db_path,
            job_id=job_id,
            tickers_with_type=tickers_with_type,
        )

        return RebuildCorpusResponse(
            job_id=job_id,
            status="started",
            ticker_count=len(tickers_with_type),
        )


    @router.get(
        "/calibration/rebuild-corpus/{job_id}",
        response_model=RebuildCorpusProgressResponse,
    )
    async def rebuild_corpus_progress_endpoint(
        job_id: str,
        db_path: str = Depends(get_db_path),
    ) -> RebuildCorpusProgressResponse:
        """LIVE-01: Poll progress of a rebuild-corpus job."""
        async with aiosqlite.connect(db_path) as conn:
            conn.row_factory = aiosqlite.Row
            row = await (
                await conn.execute(
                    """
                    SELECT job_id, status, tickers_total, tickers_completed,
                           ticker_progress_json, started_at, completed_at, error_message
                    FROM corpus_rebuild_jobs
                    WHERE job_id = ?
                    """,
                    (job_id,),
                )
            ).fetchone()

        if row is None:
            raise HTTPException(
                status_code=404,
                detail=f"Job not found: {job_id}",
            )

        return RebuildCorpusProgressResponse(
            job_id=row["job_id"],
            status=row["status"],
            tickers_total=row["tickers_total"],
            tickers_completed=row["tickers_completed"],
            ticker_progress=json.loads(row["ticker_progress_json"] or "{}"),
            started_at=row["started_at"],
            completed_at=row["completed_at"],
            error_message=row["error_message"],
        )


    async def _run_batch_rebuild(
        db_path: str,
        job_id: str,
        tickers_with_type: list[tuple[str, str]],
    ) -> None:
        """Background task: rebuild corpus per-ticker, update corpus_rebuild_jobs row.

        Per-ticker failures are captured in ticker_progress; the batch continues.
        Each ticker's rebuild_signal_corpus call inherits its own FOUND-07 two-connection
        pattern and BLOCKER 3 DELETE rollback guard — we do NOT add a second layer.

        Security: error messages from rebuild_signal_corpus are truncated to 500 chars
        before persistence (matches the daemon pattern in jobs.py line 1136) to avoid
        leaking full tracebacks into the DB.
        """
        from daemon.jobs import rebuild_signal_corpus

        progress: dict[str, dict] = {
            t: {"status": "pending"} for t, _ in tickers_with_type
        }

        # Initialize all to "pending" already done by caller; update progressively.
        successes = 0
        failures = 0

        for ticker, asset_type in tickers_with_type:
            progress[ticker] = {"status": "running"}
            await _update_progress(db_path, job_id, progress, increment_completed=False)

            try:
                result = await rebuild_signal_corpus(
                    db_path=db_path,
                    tickers=[(ticker, asset_type)],
                )
                progress[ticker] = {
                    "status": "success",
                    "rows_inserted": int(result.get("rows_inserted", 0)),
                }
                successes += 1
            except Exception as exc:
                err_text = str(exc)[:500]
                _route_logger.error(
                    "rebuild_corpus_batch: ticker=%s failed: %s",
                    ticker, err_text,
                )
                progress[ticker] = {
                    "status": "error",
                    "error": err_text,
                }
                failures += 1

            await _update_progress(db_path, job_id, progress, increment_completed=True)

        # Final status
        if failures == 0:
            final_status = "success"
        elif successes == 0:
            final_status = "error"
        else:
            final_status = "partial"

        completed_at = datetime.now(timezone.utc).isoformat()
        async with aiosqlite.connect(db_path) as conn:
            await conn.execute(
                """
                UPDATE corpus_rebuild_jobs
                SET status = ?, completed_at = ?, ticker_progress_json = ?
                WHERE job_id = ?
                """,
                (final_status, completed_at, json.dumps(progress), job_id),
            )
            await conn.commit()


    async def _update_progress(
        db_path: str,
        job_id: str,
        progress: dict[str, dict],
        increment_completed: bool,
    ) -> None:
        """Persist per-ticker progress to corpus_rebuild_jobs.

        Separate connection per call (short-lived) matches the WR-02 pattern used by
        daemon/jobs.py::prune_signal_history.
        """
        try:
            async with aiosqlite.connect(db_path) as conn:
                if increment_completed:
                    await conn.execute(
                        """
                        UPDATE corpus_rebuild_jobs
                        SET ticker_progress_json = ?,
                            tickers_completed = tickers_completed + 1
                        WHERE job_id = ?
                        """,
                        (json.dumps(progress), job_id),
                    )
                else:
                    await conn.execute(
                        """
                        UPDATE corpus_rebuild_jobs
                        SET ticker_progress_json = ?
                        WHERE job_id = ?
                        """,
                        (json.dumps(progress), job_id),
                    )
                await conn.commit()
        except Exception as upd_exc:
            # Don't let progress-update failures crash the batch
            _route_logger.warning(
                "rebuild_corpus_batch progress update failed for job_id=%s: %s",
                job_id, upd_exc,
            )
    ```
