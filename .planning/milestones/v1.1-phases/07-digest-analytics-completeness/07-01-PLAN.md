    **Sub-step 1.4 — Create dividend_cache.py (Parquet sibling cache, 24h TTL).**

    Create `data_providers/dividend_cache.py`:
    ```python
    """Parquet sibling cache for YFinance dividend history (AN-01).

    Mirrors the FOUND-02 OHLCV cache pattern (data/cache/ohlcv/) but for
    dividend Series. TTL 24h: dividends are quarterly events, so a daily
    refresh is conservative.
    """
    from __future__ import annotations

    import logging
    import os
    import time
    from datetime import date, datetime
    from pathlib import Path

    import pandas as pd

    logger = logging.getLogger(__name__)

    CACHE_DIR = Path("data") / "cache" / "dividends"
    CACHE_TTL_SECONDS = 24 * 60 * 60  # 24h

    def _cache_path(ticker: str) -> Path:
        safe = ticker.replace("/", "_").replace("\\", "_")
        return CACHE_DIR / f"{safe}.parquet"

    def read_dividend_cache(ticker: str) -> list[tuple[date, float]] | None:
        """Return cached dividends if file exists and mtime within TTL, else None."""
        path = _cache_path(ticker)
        if not path.exists():
            return None
        try:
            age = time.time() - path.stat().st_mtime
            if age > CACHE_TTL_SECONDS:
                return None
            df = pd.read_parquet(path)
            if df is None or df.empty:
                return []
            out: list[tuple[date, float]] = []
            for _, row in df.iterrows():
                try:
                    d = pd.to_datetime(row["date"]).date()
                    out.append((d, float(row["amount"])))
                except (KeyError, ValueError):
                    continue
            return out
        except Exception as exc:
            logger.warning("dividend cache read failed for %s: %s", ticker, exc)
            return None

    def write_dividend_cache(ticker: str, dividends: list[tuple[date, float]]) -> None:
        """Atomic Parquet write (FOUND-02 pattern: tmp + rename)."""
        try:
            CACHE_DIR.mkdir(parents=True, exist_ok=True)
            path = _cache_path(ticker)
            tmp = path.with_suffix(".parquet.tmp")
            df = pd.DataFrame(
                [{"date": d.isoformat(), "amount": float(a)} for d, a in dividends]
            )
            df.to_parquet(tmp, index=False)
            os.replace(tmp, path)  # atomic on Windows + POSIX
        except Exception as exc:
            logger.warning("dividend cache write failed for %s: %s", ticker, exc)
    ```

    ```python
    """Per-agent IC-IR drift detection with auto weight-scaling (AN-02).

    Run weekly by daemon Sunday 17:30 (before the 18:00 digest, so the digest
    can read drift_log for its IC-IR-movers section).

    Thresholds (RESEARCH-LOCKED, preliminary):
      - pct_drop: current IC-IR < avg_60d * (1 - 0.20)  → DRIFT_THRESHOLD_PCT=20.0
      - absolute_floor: current IC-IR < 0.5 for 2 consecutive weekly samples
                        → ICIR_FLOOR=0.5

    When total observations per agent < MIN_SAMPLES_FOR_REAL_THRESHOLD (=60),
    preliminary_threshold=1 (no trigger fires; mirrors Phase 2 preliminary_calibration).

    NEVER-ZERO-ALL guard: if all agents in an asset_type would scale to 0,
    skip the auto-scale + emit CRITICAL log + record drift_log rows with
    weight_after=NULL. Forced manual review.
    """
    from __future__ import annotations

    import logging
    from datetime import datetime, timezone
    from typing import Any

    logger = logging.getLogger("investment_agent.drift_detector")

    DRIFT_THRESHOLD_PCT: float = 20.0          # > 20% IC-IR drop triggers
    ICIR_FLOOR: float = 0.5                    # IC-IR < 0.5 for 2 weeks triggers
    MIN_SAMPLES_FOR_REAL_THRESHOLD: int = 60   # weekly IC samples per agent
    SCALE_DIVISOR: float = 2.0                 # matches compute_ic_weights

    KNOWN_AGENTS: tuple[str, ...] = (
        "TechnicalAgent",
        "FundamentalAgent",
        "MacroAgent",
        "SentimentAgent",
        "CryptoAgent",
    )
    ASSET_TYPES: tuple[str, ...] = ("stock", "btc", "eth")


    async def evaluate_drift(db_path: str) -> list[dict[str, Any]]:
        """Evaluate IC-IR drift for all (agent, asset_type) combos.

        Returns list of drift_log row dicts. Side-effects:
          - INSERT each row into drift_log
          - For triggered rows: UPSERT agent_weights with scaled weight
            (subject to NEVER-zero-all guard + WHERE manual_override=0)
          - Emit alert via PortfolioMonitor when triggered (or CRITICAL when
            never-zero guard fires)
        """
        # Deferred imports — avoids circular dep with engine/* and tracking/*
        import aiosqlite  # noqa: PLC0415
        from tracking.store import SignalStore  # noqa: PLC0415
        from tracking.tracker import SignalTracker  # noqa: PLC0415

        evaluated_at = datetime.now(timezone.utc).isoformat()
        all_rows: list[dict[str, Any]] = []

        # 1. Compute per-agent current IC-IR + preliminary flag
        async with aiosqlite.connect(db_path) as conn:
            store = SignalStore(conn)
            tracker = SignalTracker(store)

            for agent in KNOWN_AGENTS:
                _overall, rolling = await tracker.compute_rolling_ic(
                    agent, horizon="5d", window=60
                )
                valid = [ic for ic in (rolling or []) if ic is not None]
                current_icir = SignalTracker.compute_icir(valid) if valid else None
                preliminary = len(valid) < MIN_SAMPLES_FOR_REAL_THRESHOLD

                # Use drift_log history for 60d baseline (last 60 entries per agent)
                avg_60d = await _get_avg_icir_60d(conn, agent)

                delta_pct = None
                threshold_type = "none"
                triggered = False

                if not preliminary and current_icir is not None and avg_60d is not None:
                    if avg_60d != 0:
                        delta_pct = (current_icir - avg_60d) / abs(avg_60d) * 100.0
                    # Rule 1: pct_drop
                    if delta_pct is not None and delta_pct < -DRIFT_THRESHOLD_PCT:
                        triggered = True
                        threshold_type = "pct_drop"
                    # Rule 2: absolute_floor — current AND last weekly sample both < 0.5
                    elif current_icir < ICIR_FLOOR and await _prior_week_below_floor(
                        conn, agent
                    ):
                        triggered = True
                        threshold_type = "absolute_floor"

                # Rows fan out per asset_type (FundamentalAgent only stock; CryptoAgent only btc/eth)
                applicable_types = _applicable_asset_types(agent)
                for at in applicable_types:
                    all_rows.append({
                        "agent_name": agent,
                        "asset_type": at,
                        "evaluated_at": evaluated_at,
                        "current_icir": current_icir,
                        "avg_icir_60d": avg_60d,
                        "delta_pct": delta_pct,
                        "threshold_type": threshold_type,
                        "triggered": 1 if triggered else 0,
                        "preliminary_threshold": 1 if preliminary else 0,
                        "weight_before": None,   # filled in scale step
                        "weight_after": None,    # filled in scale step
                    })

        # 2. Apply auto-scale per asset_type (with NEVER-zero-all guard)
        for at in ASSET_TYPES:
            triggered_rows_for_at = [
                r for r in all_rows if r["asset_type"] == at and r["triggered"] == 1
            ]
            if not triggered_rows_for_at:
                continue
            await _apply_drift_scale(db_path, at, all_rows, triggered_rows_for_at)

        # 3. Persist all drift_log rows
        async with aiosqlite.connect(db_path) as conn:
            for r in all_rows:
                await conn.execute(
                    """
                    INSERT INTO drift_log (
                        agent_name, asset_type, evaluated_at,
                        current_icir, avg_icir_60d, delta_pct, threshold_type,
                        triggered, preliminary_threshold,
                        weight_before, weight_after
                    ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
                    """,
                    (
                        r["agent_name"], r["asset_type"], r["evaluated_at"],
                        r["current_icir"], r["avg_icir_60d"], r["delta_pct"],
                        r["threshold_type"], r["triggered"], r["preliminary_threshold"],
                        r["weight_before"], r["weight_after"],
                    ),
                )
            await conn.commit()

        # 4. Emit alert per triggered row via PortfolioMonitor (deferred import)
        for r in all_rows:
            if r["triggered"] == 1 and r["preliminary_threshold"] == 0:
                await _emit_drift_alert(db_path, r)

        return all_rows


    def _applicable_asset_types(agent: str) -> tuple[str, ...]:
        """FundamentalAgent only stock; CryptoAgent only btc/eth; rest all 3."""
        if agent == "FundamentalAgent":
            return ("stock",)
        if agent == "CryptoAgent":
            return ("btc", "eth")
        return ASSET_TYPES


    async def _get_avg_icir_60d(conn, agent: str) -> float | None:
        """Average IC-IR over last 60 weekly drift_log entries for this agent."""
        row = await (
            await conn.execute(
                """
                SELECT AVG(current_icir) FROM (
                    SELECT current_icir FROM drift_log
                    WHERE agent_name = ? AND current_icir IS NOT NULL
                    ORDER BY evaluated_at DESC
                    LIMIT 60
                )
                """,
                (agent,),
            )
        ).fetchone()
        return float(row[0]) if row and row[0] is not None else None


    async def _prior_week_below_floor(conn, agent: str) -> bool:
        """Return True if the most recent prior weekly drift_log row for `agent`
        had current_icir < ICIR_FLOOR. Used by absolute_floor rule (2 consecutive)."""
        row = await (
            await conn.execute(
                """
                SELECT current_icir FROM drift_log
                WHERE agent_name = ? AND current_icir IS NOT NULL
                ORDER BY evaluated_at DESC
                LIMIT 1
                """,
                (agent,),
            )
        ).fetchone()
        if not row or row[0] is None:
            return False
        return float(row[0]) < ICIR_FLOOR


    async def _apply_drift_scale(
        db_path: str,
        asset_type: str,
        all_rows: list[dict[str, Any]],
        triggered_rows: list[dict[str, Any]],
    ) -> None:
        """UPSERT scaled weights into agent_weights with NEVER-zero-all guard."""
        import aiosqlite  # noqa: PLC0415

        async with aiosqlite.connect(db_path) as conn:
            conn.row_factory = aiosqlite.Row
            current = {}
            existing_rows = await (
                await conn.execute(
                    "SELECT agent_name, weight, manual_override FROM agent_weights "
                    "WHERE asset_type = ?",
                    (asset_type,),
                )
            ).fetchall()
            for er in existing_rows:
                current[er["agent_name"]] = (
                    float(er["weight"]),
                    int(er["manual_override"]),
                )

            # Compute proposed new weights for ALL agents in this asset_type
            # (triggered → scale; non-triggered → keep current)
            proposed: dict[str, float] = {}
            for agent_name, (w, _mo) in current.items():
                # Find this agent's drift row (if it has one for this asset_type)
                agent_drift = next(
                    (r for r in all_rows
                     if r["agent_name"] == agent_name and r["asset_type"] == asset_type),
                    None,
                )
                if agent_drift and agent_drift["triggered"] == 1:
                    icir = agent_drift["current_icir"] or 0.0
                    factor = max(0.0, icir / SCALE_DIVISOR)
                    proposed[agent_name] = w * factor
                else:
                    proposed[agent_name] = w

            # NEVER-zero-all guard
            total = sum(proposed.values())
            if total <= 0.0:
                logger.critical(
                    "DRIFT_NEVER_ZERO: all %d agents in asset_type=%s would "
                    "scale to 0 — skipping auto-scale, manual review required.",
                    len(proposed), asset_type,
                )
                # Mark drift rows with weight_after=None
                for r in triggered_rows:
                    r["weight_before"] = current.get(r["agent_name"], (None, 0))[0]
                    r["weight_after"] = None
                # Emit CRITICAL alert via PortfolioMonitor
                await _emit_critical_drift_alert(db_path, asset_type)
                return

            # UPSERT each triggered agent's new weight (WHERE manual_override=0)
            for r in triggered_rows:
                agent_name = r["agent_name"]
                w_before, manual_override = current.get(agent_name, (0.0, 0))
                w_after = proposed[agent_name]
                if manual_override == 0:
                    await conn.execute(
                        """
                        INSERT INTO agent_weights
                            (agent_name, asset_type, weight, source, manual_override, excluded, updated_at)
                        VALUES (?, ?, ?, 'ic_ir', 0, 0, CURRENT_TIMESTAMP)
                        ON CONFLICT(agent_name, asset_type) DO UPDATE SET
                            weight = excluded.weight,
                            source = 'ic_ir',
                            updated_at = CURRENT_TIMESTAMP
                        WHERE agent_weights.manual_override = 0
                        """,
                        (agent_name, asset_type, w_after),
                    )
                    r["weight_before"] = w_before
                    r["weight_after"] = w_after
                else:
                    # Manual override active → drift_log records the would-be values, no DB write
                    r["weight_before"] = w_before
                    r["weight_after"] = w_before  # unchanged
            await conn.commit()


    async def _emit_drift_alert(db_path: str, row: dict[str, Any]) -> None:
        """Emit a WARNING-severity alert via existing monitoring pipeline."""
        from monitoring.models import Alert  # noqa: PLC0415
        from monitoring.store import AlertStore  # noqa: PLC0415

        store = AlertStore(db_path)
        alert = Alert(
            ticker="MARKET",
            alert_type="DRIFT_DETECTED",
            severity="WARNING",
            message=(
                f"{row['agent_name']} IC-IR drift on {row['asset_type']}: "
                f"current={row['current_icir']:.3f}, "
    ```

    ```python
    """Drift log endpoint (AN-02 — Phase 7).

    Frontend (07-03) reads this to render the per-row drift badge on
    CalibrationPage. Returns the LATEST drift_log row per (agent_name, asset_type)
    sorted by evaluated_at DESC.
    """
    from __future__ import annotations

    import logging
    from typing import Any

    import aiosqlite
    from fastapi import APIRouter, Depends

    from api.deps import get_db_path

    _logger = logging.getLogger(__name__)
    router = APIRouter()


    @router.get("/log")
    async def get_drift_log(
        db_path: str = Depends(get_db_path),
    ) -> dict[str, Any]:
        """Return latest drift_log row per (agent_name, asset_type)."""
        async with aiosqlite.connect(db_path) as conn:
            conn.row_factory = aiosqlite.Row
            # Self-join trick: latest evaluated_at per (agent, asset_type)
            rows = await (
                await conn.execute(
                    """
                    SELECT d.agent_name, d.asset_type, d.evaluated_at,
                           d.current_icir, d.avg_icir_60d, d.delta_pct,
                           d.threshold_type, d.triggered, d.preliminary_threshold,
                           d.weight_before, d.weight_after
                    FROM drift_log d
                    INNER JOIN (
                        SELECT agent_name, asset_type, MAX(evaluated_at) AS max_at
                        FROM drift_log
                        GROUP BY agent_name, asset_type
                    ) latest
                    ON d.agent_name = latest.agent_name
                       AND d.asset_type = latest.asset_type
                       AND d.evaluated_at = latest.max_at
                    ORDER BY d.evaluated_at DESC
                    """
                )
            ).fetchall()

        return {
            "drifts": [
                {
                    "agent_name": row["agent_name"],
                    "asset_type": row["asset_type"],
                    "evaluated_at": row["evaluated_at"],
                    "current_icir": row["current_icir"],
                    "avg_icir_60d": row["avg_icir_60d"],
                    "delta_pct": row["delta_pct"],
                    "threshold_type": row["threshold_type"],
                    "triggered": bool(row["triggered"]),
                    "preliminary_threshold": bool(row["preliminary_threshold"]),
                    "weight_before": row["weight_before"],
                    "weight_after": row["weight_after"],
                }
                for row in rows
            ]
        }
    ```
